sudo firewall-cmd --get-active-zones

# 開放 TCP 5432 port（例如 PostgreSQL）
sudo firewall-cmd --permanent --add-port=5432/tcp

# 關閉 TCP 5432 port
sudo firewall-cmd --permanent --remove-port=5432/tcp

# 只開放特定ip連入特定port
sudo firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="192.168.1.101" port protocol="tcp" port="5432" accept'

# 移除 service 的 ssh（避免所有人都能連）
sudo firewall-cmd --permanent --remove-service=ssh
# 只開放特定ip連入特定服務
sudo firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="192.168.1.10" service name="ssh" accept'

sudo firewall-cmd --reload
